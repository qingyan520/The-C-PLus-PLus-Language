# 二叉搜索树









```cpp
#include<iostream>
using namespace std;
template<class K>
struct BSTNode
{
    BSTNode<K>*_left;
    BSTNode<K>*_right;
    
    K _key;
}
template<class K>
class BST
{
  typedef BSTNode<K> Node;
    public:
    BST():root(nullptr)
    {
        
    }
    bool Insert(const K&key)
    {
        if(_root==nullptr)
        {
            _root=new Node(key);
            return true;
        }
        Node*cur=_root;
        Node*parent=nullptr;
        while(cur)
        {
            parent=cur;
            if(key>cur->_K)
            {
                cur=cur->_right;
            }
            else if(key<cur->_K)
            {
                cur=cur->left;
            }
            else
            {
                return false;
               
            }
        }
        
        cur=new Node(key);
        if(parent->_K>key)
        {
            parent->left=cur;
        }
        else
        {
            parent->right=cur;
        }
        return true;  
    }
    
    
    
    
    
    private:
    
    Node*_root;
    
};

}
```
